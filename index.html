/* ----- DYNAMIC LEADERBOARDS ----- */ function updateLeaderboards(){ // MVP Leaderboard let mvpTable = document.querySelector("#mvp-leaderboard tbody"); mvpTable.innerHTML=""; let sortedMVP = Object.keys(users) .filter(u=>!users[u].isAdmin) .sort((a,b)=>users[b].mvpPoints - users[a].mvpPoints); sortedMVP.forEach((u,i)=>{ let row = document.createElement("tr"); row.innerHTML = `${i+1}${u}${users[u].mvpPoints}`; mvpTable.appendChild(row); }); // Coin Leaderboard let coinTable = document.querySelector("#coin-leaderboard tbody"); coinTable.innerHTML=""; let sortedCoins = Object.keys(users) .filter(u=>!users[u].isAdmin) .sort((a,b)=>users[b].totalCoins - users[a].totalCoins); sortedCoins.forEach((u,i)=>{ let row = document.createElement("tr"); row.innerHTML = `${i+1}${u}${users[u].totalCoins}`; coinTable.appendChild(row); }); } /* ----- INITIAL UPDATE ----- */ updateLeaderboards(); updateUI(); /* ----- REFRESH SESSION HANDLING ----- */ window.addEventListener('beforeunload',()=>{ // Save current user balances and bets to localStorage for persistence localStorage.setItem("kickcoinsUsers", JSON.stringify(users)); }); window.addEventListener('load',()=>{ let saved = localStorage.getItem("kickcoinsUsers"); if(saved){ users = JSON.parse(saved); updateLeaderboards(); updateUI(); } });
